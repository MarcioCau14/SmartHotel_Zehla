    67→// ── Função Principal: Aprender de uma Conversa ────────────────────────
    68→
    69→/**
    70→ * Analisa uma conversa recém-resolvida e extrai padrões aprendidos.
    71→ * Roda em background (não bloqueia a resposta ao hóspede).
    72→ *
    73→ * Triage (do self-learning-skills):
    74→ * - Pergunta + resposta completa → KnowledgeEntry (padrão reutilizável)
    75→ * - Facto solto (ex: "WiFi é Pousada2024") → ignorado (muda demais)
    76→ * - One-off (ex: "ônibus 834 sai 3x/dia") → ignorado (não é recorrente)
    77→ */
    78→export async function learnFromConversation(
    79→  tenantId: string,
    80→  conversationId: string,
    81→): Promise<LearningResult> {
    82→  const result: LearningResult = {
    83→    patternsExtracted: 0,
    84→    patternsSkipped: 0,
    85→    patternsPromoted: 0,
    86→    patternsDeprecated: 0,
    87→    details: [],
    88→  };
    89→
    90→  try {
    91→    // 1. Buscar todas as mensagens da conversa
    92→    const messages = await db.conversationMessage.findMany({
    93→      where: { conversationId },
    94→      orderBy: { timestamp: 'asc' },
    95→    });
    96→
    97→    if (messages.length < MIN_CONVERSATION_MESSAGES) {
    98→      result.details.push(`Conversa curta (${messages.length} msgs), pulando.`);
    99→      return result;
   100→    }
   101→
   102→    // 2. Buscar status da conversa
   103→    const conversation = await db.conversationLog.findUnique({
   104→      where: { id: conversationId },
   105→      select: { status: true, aiConfidence: true },
   106→    });
   107→
   108→    if (!conversation) return result;
   109→
   110→    const isResolved = conversation.status === 'resolved';
   111→    const isEscalated = conversation.status === 'escalated';
   112→
   113→    // v2.0: Analisar sentimento do hóspede na conversa
   114→    const guestMessages = messages.filter(m => m.from === 'guest').map(m => m.content);
   115→    const sentimentScore = analyzeConversationSentiment(guestMessages);
   116→
   117→    // 3. Extrair pares pergunta (hóspede) → resposta (IA)
   118→    const pairs: Array<{ guestMsg: string; aiMsg: string }> = [];
   119→    for (let i = 0; i < messages.length - 1; i++) {
   120→      if (messages[i].from === 'guest' && messages[i + 1].from === 'ai') {
   121→        pairs.push({
   122→          guestMsg: messages[i].content,
   123→          aiMsg: messages[i + 1].content,
   124→        });
   125→      }
   126→    }
   127→
   128→    if (pairs.length === 0) {
   129→      result.details.push('Nenhum par hóspede→IA encontrado.');
   130→      return result;
   131→    }
   132→
   133→    // 4. Buscar conhecimento existente para evitar duplicatas
   134→    const existingKnowledge = await db.knowledgeEntry.findMany({
   135→      where: { tenantId },
   136→      select: { question: true },
   137→    });
   138→    const existingQuestions = new Set(
   139→      existingKnowledge.map(e => e.question.toLowerCase().trim())
   140→    );
   141→
   142→    // 5. Extrair padrões candidatos (apenas de conversas resolvidas)
   143→    if (isResolved) {
   144→      const candidates = extractCandidatePatterns(pairs);
   145→
   146→      for (const candidate of candidates) {
   147→        if (result.patternsExtracted >= MAX_PATTERNS_PER_RUN) break;
   148→
   149→        // Triage: pular fatos soltos e one-offs
   150→        if (!isReusablePattern(candidate.question)) {
   151→          result.patternsSkipped++;
   152→          result.details.push(`Triage SKIP (não reutilizável): "${candidate.question.substring(0, 50)}..."`);
   153→          continue;
   154→        }
   155→
   156→        // Deduplicação: verificar se já existe algo similar
   157→        if (isDuplicateQuestion(candidate.question, existingQuestions)) {
   158→          result.patternsSkipped++;
   159→          result.details.push(`DUPLICATA: "${candidate.question.substring(0, 50)}..."`);
   160→          continue;
   161→        }
   162→
   163→        // Sanitização: remover PII e conteúdo sensível
   164→        const sanitized = sanitizePattern(candidate);
   165→        if (!sanitized) {
   166→          result.patternsSkipped++;
   167→          continue;
   168→        }
   169→
   170→        // v2.0: Sentiment-weighted initial confidence
   171→        // Conversas com sentimento positivo geram padrões com confiança levemente maior
   172→        const sentimentAdjustedConfidence = Math.min(
   173→          1.0,
   174→          INITIAL_CONFIDENCE + (sentimentScore > 0 ? sentimentScore * 0.1 : 0)
   175→        );
   176→
   177→        // Criar novo KnowledgeEntry
   178→        try {
   179→          const newEntry = await db.knowledgeEntry.create({
   180→            data: {
   181→              tenantId,
   182→              question: sanitized.question,
   183→              answer: sanitized.answer,
   184→              category: sanitized.category,
   185→              priority: 'medium', // Precisa ser promovido para 'high'
   186→              usage: 0,
   187→              effectiveness: sentimentAdjustedConfidence * 100, // 0-100 scale
   188→              embeddingJson: '[]', // Será vetorizado pelo RAG na próxima busca
   189→              metadata: JSON.stringify({
   190→                source: 'auto_learned',
   191→                sourceConversationId: conversationId,
   192→                extractedAt: new Date().toISOString(),
   193→                confidence: sentimentAdjustedConfidence,
   194→                timesUsed: 0,
   195→                timesSuccessful: 0,
   196→                verified: false,
   197→                sentimentScore, // v2.0: rastreia sentimento da conversa de origem
   198→              }),
   199→            },
   200→          });
   201→
   202→          result.patternsExtracted++;
   203→          result.details.push(`NOVO padrão (sentimento: ${sentimentScore.toFixed(2)}): "${sanitized.question.substring(0, 60)}..."`);
   204→          existingQuestions.add(sanitized.question.toLowerCase().trim());
   205→
   206→          // Log de atividade de aprendizado
   207→          await db.aIActivityLog.create({
   208→            data: {
   209→              tenantId,
   210→              type: 'learning',
   211→              message: `Padrão aprendido da conversa ${conversationId}: "${sanitized.question.substring(0, 80)}"`,
   212→              status: 'success',
   213→              duration: 0,
   214→              metadata: JSON.stringify({
   215→                knowledgeEntryId: newEntry.id,
   216→                sourceConversationId: conversationId,
   217→                patternType: 'auto_extracted',
   218→                sentimentScore, // v2.0
   219→              }),
   220→            },
   221→          });
   222→        } catch (err) {
   223→          console.error('[ConversationLearner] Falha ao criar KnowledgeEntry:', err);
   224→        }
   225→      }
   226→    }
   227→
   228→    // v2.0: Negative Knowledge Capture (inspirado em self-learning-skills Pattern 7)
   229→    // Quando uma conversa é escalada, captura o que a IA fez ERRADO como anti-padrão
   230→    if (isEscalated && pairs.length > 0) {
   231→      const antiPatternsResult = await captureAntiPatterns(tenantId, conversationId, pairs);
   232→      result.details.push(...antiPatternsResult.details);
   233→    }
   234→
   235→    // 6. Feedback loop: atualizar effectiveness de knowledge usado nesta conversa
   236→    // v2.0: Agora com sentiment-weighted feedback
   237→    if (pairs.length > 0) {
   238→      await updateKnowledgeEffectiveness(
   239→        tenantId,
   240→        pairs[pairs.length - 1].guestMsg,
   241→        isResolved,
   242→        isEscalated,
   243→        sentimentScore, // v2.0: sentimento pondera o feedback
   244→      );
   245→    }
   246→
   247→    // 7. Promover padrões que atingiram o threshold (Promotion Rule)
   248→    const promotionResult = await promoteVerifiedPatterns(tenantId);
   249→    result.patternsPromoted = promotionResult.promoted;
   250→    result.patternsDeprecated = promotionResult.deprecated;
   251→    result.details.push(...promotionResult.details);
   252→
   253→    // v2.0: Confidence Decay — decair padrões não usados recentemente
   254→    const decayResult = await applyConfidenceDecay(tenantId);
   255→    if (decayResult.decayed > 0) {
   256→      result.details.push(`DECAY: ${decayResult.decayed} padrões sofreram decaimento por inatividade`);
   257→    }
   258→
   259→    // 8. Atualizar contadores no AgentConfig
   260→    await updateAgentConfigLearningStats(tenantId, result);
   261→
   262→  } catch (error) {
   263→    console.error('[ConversationLearner] Erro em learnFromConversation:', error);
   264→    result.details.push(`ERRO: ${error instanceof Error ? error.message : 'desconhecido'}`);
   265→  }
   266→
   267→  return result;
   268→}

   270→// ── Extração de Padrões Candidatos ────────────────────────────────────
   271→
   272→function extractCandidatePatterns(
   273→  pairs: Array<{ guestMsg: string; aiMsg: string }>,
   274→): ExtractedPattern[] {
   275→  const candidates: ExtractedPattern[] = [];
   276→
   277→  for (const pair of pairs) {
   278→    const question = pair.guestMsg.trim();
   279→    const answer = pair.aiMsg.trim();
   280→
   281→    // Filtrar perguntas muito curtas ou sem substância
   282→    if (question.length < 10) continue;
   283→    // Filtrar respostas muito curtas
   284→    if (answer.length < 20) continue;
   285→    // Filtrar respostas de erro / fallback
   286→    if (answer.includes('probleminha para processar') || answer.includes('silenciada')) continue;
   287→    // Filtrar mensagens de captura de dados
   288→    if (answer.includes('Me passa seu e-mail') || answer.includes('WhatsApp é o melhor número')) continue;
   289→
   290→    candidates.push({
   291→      question,
   292→      answer,
   293→      category: classifyCategory(question),
   294→      sourceConversationId: '',
   295→      confidence: INITIAL_CONFIDENCE,
   296→    });
   297→  }
   298→
   299→  return candidates;
   300→}

   304→function isReusablePattern(question: string): boolean {
   305→  const q = question.toLowerCase();
   306→
   307→  // Padrões reutilizáveis (perguntas que outros hóspedes também fariam)
   308→  const reusablePatterns = [
   309→    /pre[çc]o|valor|custo|quanto\s+custa|tarifa/,
   310→    /dispon[ií]vel|vaga|tem\s+quarto|tem\s+chal[eé]/,
   311→    /check[\s-]?in|check[\s-]?out|entrada|sa[íi]da/,
   312→    /caf[ée]\s+da\s+manh[ãa]|cafe|morning/,
   313→    /piscina|praia|piscina/,
   314→    /aceita\s+pet|cachorro|gato|animal/,
   315→    /estacionamento|carro|garagem/,
   316→    /wi[-]?fi|internet|senha/,
   317→    /cancel|desistir|reembolso/,
   318→    /como\s+chegar|localiza[çc][ãaão]|endere[çc]o|mapa/,
   319→    /crian[çc]a|bebe|ber[çc]o/,
   320→    /pagamento|pix|cart[ãa]o|parcela/,
   321→    /_transfer|aeroporto|onibus|[ôo]nibus|transporte/,
   322→    /avalia[çc][ãaõ]|review|nota/,
   323→  ];
   324→
   325→  return reusablePatterns.some(pattern => pattern.test(q));
   326→}

   328→// ── Deduplicação ──────────────────────────────────────────────────────
   329→
   330→function isDuplicateQuestion(
   331→  question: string,
   332→  existingQuestions: Set<string>,
   333→): boolean {
   334→  const normalized = question.toLowerCase().trim();
   335→
   336→  // Verificar similaridade com perguntas existentes
   337→  for (const existing of existingQuestions) {
   338→    // Verificação exata
   339→    if (existing === normalized) return true;
   340→
   341→    // Verificação de sobreposição significativa (>70% das palavras)
   342→    const existingWords = new Set(existing.split(/\s+/));
   343→    const newWords = new Set(normalized.split(/\s+/));
   344→    let overlap = 0;
   345→    for (const word of newWords) {
   346→      if (existingWords.has(word)) overlap++;
   347→    }
   348→    const similarity = overlap / Math.max(newWords.size, existingWords.size);
   349→    if (similarity > 0.7) return true;
   350→  }
   351→
   352→  return false;
   353→}

   355→// ── Sanitização de PII ───────────────────────────────────────────────
   356→
   357→function sanitizePattern(
   358→  pattern: ExtractedPattern,
   359→): ExtractedPattern | null {
   360→  let { question, answer } = pattern;
   361→
   362→  // Remover nomes próprios comuns de hóspedes (padrão: "Olá [Nome],")
   363→  question = question.replace(/^Olá\s+\w+[,!]/i, 'Olá,');
   364→  answer = answer.replace(/\b(?:Bernardo|Carlos|Maria|Ana|João|Pedro|Lucas)\b(?=,|\s+seja)/g, '[hóspede]');
   365→
   366→  // Remover números de telefone
   367→  question = question.replace(/\(?\d{2}\)?\s?\d{4,5}[-\s]?\d{4}/g, '[telefone]');
   368→  answer = answer.replace(/\(?\d{2}\)?\s?\d{4,5}[-\s]?\d{4}/g, '[telefone]');
   369→
   370→  // Remover e-mails
   371→  question = question.replace(/[\w.-]+@[\w.-]+\.\w+/g, '[email]');
   372→  answer = answer.replace(/[\w.-]+@[\w.-]+\.\w+/g, '[email]');
   373→
   374→  // Remover chaves PIX reais
   375→  answer = answer.replace(/00020101\d{20,}/g, '[PIX_COPIA_E_COLA]');
   376→
   377→  // Se ficou muito curto após sanitização, descartar
   378→  if (question.length < 10 || answer.length < 20) return null;
   379→
   380→  return { ...pattern, question, answer };
   381→}

   383→// ── Classificação de Categoria ───────────────────────────────────────
   384→
   385→function classifyCategory(question: string): string {
   386→  const q = question.toLowerCase();
   387→
   388→  if (/pre[çc]o|valor|custo|quanto\s+custa|tarifa|pagamento|pix|cart[ãa]o|parcela/.test(q)) return 'pricing';
   389→  if (/quarto|chal[eé]|su[ií]te|cama|acomoda[çc][ãaão]|dispon[ií]vel|vaga/.test(q)) return 'rooms';
   390→  if (/piscina|caf[ée]|servi[çc]o|restaurante|limpeza|arruma[çc][ãaão]|toalha|roupa/.test(q)) return 'amenities';
   391→  if (/check[\s-]?in|check[\s-]?out|pol[ií]tica|regra|hor[aá]rio|cancel|reembolso/.test(q)) return 'policies';
   392→  if (/como\s+chegar|localiza[çc][ãaão]|endere[çc]o|mapa|dist[aâ]ncia|transfer|aeroporto|onibus|transporte/.test(q)) return 'location';
   393→  if (/passeio|tour|atividade|praia|trilha|passear|vizitar/.test(q)) return 'activities';
   394→  if (/cafe|caf[ée]\s+da\s+manh[ãa]|jantar|almo[çc]o|restaurante|comida/.test(q)) return 'food';
   395→
   396→  return 'custom';
   397→}

   399→// ── v2.0: Sentiment Analysis (keyword-based, zero dependências) ──────
   400→
   401→/**
   402→ * Analisa o sentimento geral das mensagens do hóspede.
   403→ * Retorna score entre -1 (muito negativo) e +1 (muito positivo).
   404→ * Zero-dependency: usa apenas regex, nenhum modelo externo.
   405→ * Inspirado na abordagem de sentiment analysis do awesome-machine-learning.
   406→ */
   407→function analyzeConversationSentiment(guestMessages: string[]): number {
   408→  if (guestMessages.length === 0) return 0;
   409→
   410→  const POSITIVE_PATTERNS = [
   411→    /obrigad[oa]/i, /valeu/i, /perfeito/i, /excelente/i, /maravilh[oaos]/i,
   412→    /adorei/i, /amei/i, /massa/i, /top/i, /legal/i, /bonito/i,
   413→    /lindo/i, /gostei/i, /recomend[oaos]/i, /voltar/i, /melhor/i,
   414→    /ótimo/i, /otimo/i, /show/i, /incr[íi]vel/i, /surpreend/i,
   415→    /bem/i, /sim/i, /claro/i, /com certeza/i, /beleza/i,
   416→  ];
   417→
   418→  const NEGATIVE_PATTERNS = [
   419→    /p[ée]ssimo/i, /horr[íi]vel/i, /ruim/i, /n[aã]o gostei/i, /desapont/i,
   420→    /problema/i, /erro/i, /errado/i, /nao funciona/i, /n[aã]o funciona/i,
   421→    /reclama[çc][aã]o/i, /insatisfeit[oa]/i, /frustrad[oa]/i, /pessimo/i,
   422→    /engana[çc][aã]o/i, /mentira/i, /falso/i, /desistir/i, /cancelar/i,
   423→    /n[aã]o recomendo/i, /nunca mais/i, /dinheiro/i, /caro/i,
   424→  ];
   425→
   426→  let totalScore = 0;
   427→  let scoredMessages = 0;
   428→
   429→  // Analisar apenas as últimas 5 mensagens (mais relevantes para o outcome)
   430→  const recentMessages = guestMessages.slice(-5);
   431→
   432→  for (const msg of recentMessages) {
   433→    let msgScore = 0;
   434→    let hasSignal = false;
   435→
   436→    for (const pattern of POSITIVE_PATTERNS) {
   437→      if (pattern.test(msg)) { msgScore += 0.5; hasSignal = true; }
   438→    }
   439→    for (const pattern of NEGATIVE_PATTERNS) {
   440→      if (pattern.test(msg)) { msgScore -= 0.5; hasSignal = true; }
   441→    }
   442→
   443→    // Negation detection: "não gostei" → inverte positivo, "não é ruim" → inverte negativo
   444→    if (/n[aã]o\s+(é|gostei|recomendo|voltaria)/i.test(msg)) {
   445→      msgScore = -Math.abs(msgScore) || -0.3;
   446→      hasSignal = true;
   447→    }
   448→
   449→    if (hasSignal) {
   450→      totalScore += Math.max(-1, Math.min(1, msgScore));
   451→      scoredMessages++;
   452→    }
   453→  }
   454→
   455→  return scoredMessages > 0 ? totalScore / scoredMessages : 0;
   456→}

   458→// ── v2.0: Negative Knowledge Capture ─────────────────────────────────
   459→
   460→/**
   461→ * Captura anti-padrões de conversas escaladas.
   462→ * Quando a IA falha e a conversa é escalada para humano, extrai
   463→ * o que a IA respondeu de errado para evitar repetir.
   464→ * Inspirado no Pattern 7 (Negative Knowledge) do self-learning-skills.
   465→ */
   466→async function captureAntiPatterns(
   467→  tenantId: string,
   468→  conversationId: string,
   469→  pairs: Array<{ guestMsg: string; aiMsg: string }>,
   470→): Promise<{ details: string[] }> {
   471→  const details: string[] = [];
   472→  let captured = 0;
   473→
   474→  for (const pair of pairs) {
   475→    if (captured >= ANTI_PATTERN_MAX) break;
   476→
   477→    // Filtrar respostas genéricas ou muito curtas
   478→    if (pair.aiMsg.length < 30) continue;
   479→    // Filtrar respostas de fallback/erro
   480→    if (pair.aiMsg.includes('probleminha para processar') || pair.aiMsg.includes('silenciada')) continue;
   481→
   482→    // Sanitizar PII do anti-padrão
   483→    const sanitized = sanitizePattern({
   484→      question: pair.guestMsg,
   485→      answer: pair.aiMsg,
   486→      category: 'anti_pattern',
   487→      sourceConversationId: conversationId,
   488→      confidence: 0,
   489→    });
   490→    if (!sanitized) continue;
   491→
   492→    try {

   540→// ── v2.0: Confidence Decay ───────────────────────────────────────────
   541→
   542→/**
   543→ * Aplica decaimento temporal de confiança em padrões não usados.
   544→ * Inspirado em online-learning (River) e continual-learning.
   545→ * Padrões que ninguém pergunta há 30+ dias perdem confiança gradualmente.
   546→ * Padrões usados recentemente ganham um pequeno boost de recência.
   547→ */
   548→async function applyConfidenceDecay(
   549→  tenantId: string,
   550→): Promise<{ decayed: number; boosted: number }> {
   551→  const result = { decayed: 0, boosted: 0 };
   552→
   553→  try {
   554→    const patterns = await db.knowledgeEntry.findMany({
   555→      where: {
   556→        tenantId,
   557→        category: 'auto_learned',
   558→        priority: { in: ['medium', 'high'] },
   559→      },
   560→    });
   561→
   562→    const now = Date.now();
   563→    const DAY_MS = 24 * 60 * 60 * 1000;
   564→
   565→    for (const pattern of patterns) {
   566→      const metadata = safeParseMetadata(pattern.metadata);
   567→      const { confidence = INITIAL_CONFIDENCE } = metadata;
   568→      const lastUsed = pattern.lastUsed?.getTime() || pattern.createdAt.getTime();
   569→      const daysSinceUsed = (now - lastUsed) / DAY_MS;
   570→
   571→      let newConfidence = confidence;
   572→      let changed = false;
   573→
   574→      if (daysSinceUsed > DECAY_DAYS_THRESHOLD) {
   575→        // Aplicar decay: -0.02 por dia acima do threshold
   576→        const excessDays = daysSinceUsed - DECAY_DAYS_THRESHOLD;
   577→        const decay = excessDays * DECAY_RATE_PER_DAY;
   578→        newConfidence = Math.max(DECAY_MIN_CONFIDENCE, confidence - decay);
   579→        if (newConfidence < confidence) {
   580→          changed = true;
   581→          result.decayed++;
   582→        }
   583→      } else if (daysSinceUsed <= 1 && pattern.usage > 0) {
   584→        // Boost de recência para padrões usados nas últimas 24h
   585→        newConfidence = Math.min(1, confidence + RECENCY_BOOST);
   586→        changed = true;
   587→        result.boosted++;
   588→      }
   589→
   590→      if (changed) {
   591→        const newEffectiveness = Math.min(100, newConfidence * 100);
   592→        await db.knowledgeEntry.update({
   593→          where: { id: pattern.id },
   594→          data: {
   595→            effectiveness: newEffectiveness,
   596→            metadata: JSON.stringify({
   597→              ...metadata,
   598→              confidence: newConfidence,
   599→              lastDecayAt: new Date().toISOString(),
   600→              daysSinceUsed: Math.round(daysSinceUsed),
   601→            }),
   602→          },
   603→        });
   604→      }
   605→    }
   606→  } catch (err) {
   607→    console.error('[ConversationLearner] Falha no confidence decay:', err);
   608→  }
   609→
   610→  return result;
   611→}

   613→// ── Feedback Loop: Atualizar Effectiveness (v2.0: sentiment-weighted) ─
   614→
   615→/**
   616→ * Quando uma conversa usa RAG e termina, atualiza a effectiveness
   617→ * das knowledge entries baseado no outcome.
   618→ * v2.0: Sentiment pondera o boost/penalty — hóspede satisfeito boosta mais,
   619→ * hóspede insatisfeito penaliza mais.
   620→ */
   621→async function updateKnowledgeEffectiveness(
   622→  tenantId: string,
   623→  lastGuestQuestion: string,
   624→  wasResolved: boolean,
   625→  wasEscalated: boolean,
   626→  sentimentScore: number = 0, // v2.0: default neutro
   627→): Promise<void> {
   628→  try {
   629→    // Buscar knowledge entries que foram usados recentemente
   630→    const recentEntries = await db.knowledgeEntry.findMany({
   631→      where: {
   632→        tenantId,
   633→        lastUsed: { gte: new Date(Date.now() - 5 * 60 * 1000) }, // Usados nos últimos 5 min
   634→        category: 'auto_learned',
   635→      },
   636→    });
   637→
   638→    if (recentEntries.length === 0) return;
   639→
   640→    // v2.0: Sentiment weighting factor
   641→    // sentimento > 0 (positivo) → multiplica boost por 1.0 + sentimento
   642→    // sentimento < 0 (negativo) → multiplica penalty por 1.0 + |sentimento|
   643→    const sentimentBoostMultiplier = 1 + Math.max(0, sentimentScore) * 0.5;
   644→    const sentimentPenaltyMultiplier = 1 + Math.max(0, -sentimentScore) * 0.5;
   645→
   646→    for (const entry of recentEntries) {
   647→      const metadata = safeParseMetadata(entry.metadata);
   648→      let { timesUsed = 0, timesSuccessful = 0, confidence = INITIAL_CONFIDENCE } = metadata;
   649→
   650→      timesUsed++;
   651→      if (wasResolved) {
   652→        timesSuccessful++;
   653→        // v2.0: Sentiment-weighted boost
   654→        confidence = Math.min(1, confidence + (CONFIDENCE_BOOST * sentimentBoostMultiplier));
   655→      } else if (wasEscalated) {
   656→        // v2.0: Sentiment-weighted penalty
   657→        confidence = Math.max(0, confidence - (CONFIDENCE_PENALTY * sentimentPenaltyMultiplier));
   658→      }
   659→
   660→      // Atualizar effectiveness como média ponderada
   661→      const effectiveness = timesUsed > 0
   662→        ? Math.min(100, (timesSuccessful / timesUsed) * 100)
   663→        : confidence * 100; // 0-100 scale
   664→
   665→      await db.knowledgeEntry.update({
   666→        where: { id: entry.id },
   667→        data: {
   668→          effectiveness,
   669→          metadata: JSON.stringify({
   670→            ...metadata,
   671→            timesUsed,
   672→            timesSuccessful,
   673→            confidence,
   674→            lastOutcome: wasResolved ? 'resolved' : wasEscalated ? 'escalated' : 'active',
   675→            lastOutcomeAt: new Date().toISOString(),
   676→            lastSentimentScore: sentimentScore, // v2.0: rastreia sentimento do último feedback
   677→          }),
   678→        },
   679→      });
   680→    }
   681→  } catch (err) {
   682→    console.error('[ConversationLearner] Falha ao atualizar effectiveness:', err);
   683→  }
   684→}

   686→// ── Promotion Rule: Promover/Desativar Padrões ───────────────────────
   687→
   688→async function promoteVerifiedPatterns(
   689→  tenantId: string,
   690→): Promise<{ promoted: number; deprecated: number; details: string[] }> {
   691→  const result = { promoted: 0, deprecated: 0, details: [] as string[] };
   692→
   693→  try {
   694→    // Buscar padrões auto_learned que podem ser promovidos
   695→    const candidates = await db.knowledgeEntry.findMany({
   696→      where: { tenantId, category: 'auto_learned' },
   697→    });
   698→
   699→    for (const entry of candidates) {
   700→      const metadata = safeParseMetadata(entry.metadata);
   701→      const { timesUsed = 0, timesSuccessful = 0, confidence = INITIAL_CONFIDENCE, verified = false } = metadata;
   702→
   703→      // Promotion Rule: 3 gates
   704→      if (
   705→        !verified &&
   706→        confidence >= PROMOTION_SUCCESS_THRESHOLD &&
   707→        timesUsed >= PROMOTION_MIN_USES &&
   708→        timesSuccessful >= PROMOTION_MIN_SUCCESSFUL
   709→      ) {
   710→        // PROMOVER: elevar prioridade e marcar como verificado
   711→        await db.knowledgeEntry.update({
   712→          where: { id: entry.id },
   713→          data: {
   714→            priority: 'high', // Promovido
   715→            metadata: JSON.stringify({
   716→              ...metadata,
   717→              verified: true,
   718→              verifiedAt: new Date().toISOString(),
   719→            }),
   720→          },
   721→        });
   722→        result.promoted++;
   723→        result.details.push(`PROMOVIDO: "${entry.question.substring(0, 50)}..." (conf: ${confidence.toFixed(2)}, usos: ${timesUsed})`);
   724→      }
   725→
   726→      // Deprecation: desativar padrões com baixa performance
   727→      if (
   728→        timesUsed >= DEPRECATION_MIN_USES &&
   729→        confidence < DEPRECATION_CONFIDENCE
   730→      ) {
   731→        await db.knowledgeEntry.update({
   732→          where: { id: entry.id },
   733→          data: {
   734→            priority: 'low', // Desativado pelo feedback loop
   735→            metadata: JSON.stringify({
   736→              ...metadata,
   737→              deprecated: true,
   738→              deprecatedAt: new Date().toISOString(),
   739→              deprecatedReason: `Confiança baixa (${confidence.toFixed(2)}) após ${timesUsed} usos`,
   740→            }),
   741→          },
   742→        });
   743→        result.deprecated++;
   744→        result.details.push(`DESATIVADO: "${entry.question.substring(0, 50)}..." (conf: ${confidence.toFixed(2)})`);
   745→      }
   746→    }
   747→  } catch (err) {
   748→    console.error('[ConversationLearner] Falha na promoção:', err);
   749→  }
   750→
   751→  return result;
   752→}

   934→function calculateLearningVelocity(
