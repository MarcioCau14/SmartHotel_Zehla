    31→/**
    32→ * O "modo" do tenant — define qual Strategy o orquestrador usa.
    33→ *
    34→ * No código atual, Property.type suporta: pousada|hotel|hostel|chalé|resort
    35→ * Para o AirB, precisamos adicionar: apartamento|casa|studio|loft
    36→ *
    37→ * Mas MAIS IMPORTANTE que o tipo de imóvel é o MODO DE OPERAÇÃO:
    38→ * - "pousada" = Zélla Pousada (secretária vendedora)
    39→ * - "airbnb"  = Zélla AirB (anfitrião que sabe tudo)

    41→export type OperatingMode = 'pousada' | 'airbnb';
    42→

    48→ *   - NÃO tem latitude/longitude

    50→ *   - NÃO tem houseRules

    64→  neighborhood: string;

    66→  state: string;

    68→  longitude?: number;

    89→
    90→  // Dicas do bairro (NOVO — Property NÃO tem)
    91→  neighborhoodTips: NeighborhoodTip[];

   368→/**
   369→ * Classifica a intenção de uma mensagem de hóspede Airbnb.
   370→ *
   371→ * Mesma estrutura do intent-router.ts (heurística → LLM → fallback),
   372→ * mas com o set de intenções e padrões do AirB.
   373→ */
   374→export function classifyAirBIntent(message: string): { intent: AirBIntent; confidence: number; method: 'heuristic' | 'llm' | 'fallback' } {
   375→  const lower = message.toLowerCase().trim();
   376→
   377→  // Passo 1: Heurística rápida (igual ao intent-router.ts:138-253)
   378→  for (const { intent, patterns } of AIRB_INTENT_PATTERNS) {
   379→    for (const { regex, confidence } of patterns) {
   380→      if (regex.test(lower)) {
   381→        return { intent, confidence, method: 'heuristic' };
   382→      }
   383→    }
   384→  }
   385→
   386→  // Passo 2: Em produção, chamaria LLM aqui (como intent-router.ts:256-284)
   387→  // Por ora, fallback para desconhecido
   388→  return { intent: 'UNKNOWN', confidence: 0.3, method: 'fallback' };
   389→}

   393→
   394→/**
   395→ * Ferramentas do Zélla AirB — completamente diferentes do Zélla Pousada.
   396→ *
   397→ * Comparação:
   398→ *   Zélla Pousada (zehla-tools.ts):
   399→ *     zehla_analisar_ocupacao  → Analisa taxa de ocupação
   400→ *     zehla_sugerir_preco      → Sugere preço ótimo
   401→ *     zehla_analisar_reviews   → Analisa reviews
   402→ *     zehla_gerar_relatorio_diario → Relatório operacional diário
   403→ *     zehla_buscar_dados_property  → Dados cadastrais da pousada
   404→ *
   405→ *   Zélla AirB (este arquivo):
   406→ *     airb_get_checkin_guide   → Guia completo de check-in
   407→ *     airb_get_house_rules     → Regras da casa
   408→ *     airb_get_wifi_info       → Dados do WiFi
   409→ *     airb_get_equipment_help  → Como usar equipamentos
   410→ *     airb_get_neighborhood_tips → Dicas do bairro
   411→ *     airb_get_emergency_info  → Contatos de emergência
   412→ *     airb_get_host_knowledge  → Conhecimento do anfitrião sobre o imóvel
   413→ */
   414→export interface AirBToolDefinition {
   415→  name: string;
   416→  description: string;
   417→  parameters: Record<string, any>;
   418→  execute: (args: Record<string, any>, context: AirbnbPropertyContext) => Promise<string>;
   419→}

   603→/**
   604→ * Constrói o system prompt do Zélla AirB.
   605→ *
   606→ * Comparação com o código existente:
   607→ *
   608→ *   PromptBuilder.ts (brain/processors/PromptBuilder.ts:39-53):
   609→ *     "Você é o assistente virtual da ${property.name}.
   610→ *      Atende pelo WhatsApp de forma calorosa e eficiente.
   611→ *      REGRAS: Sempre gentil, use emojis com moderação,
   612→ *              NÃO negocie preços, NÃO faça estornos."
   613→ *
   614→ *   whatsapp-ai-responder.ts (linha 363-380):
   615→ *     "Você é a ZÉLLA, uma assistente virtual de inteligência artificial
   616→ *      ultra-atenciosa e hospitaleira da pousada X.
   617→ *      Seu objetivo é sanar dúvidas, encantar o hóspede, sugerir acomodações
   618→ *      e incentivar a reserva direta de forma natural, educada e calorosa."
   619→ *
   620→ *   *** NENHUM dos dois serve para Airbnb. O tom de vendedor é tóxico. ***
   621→ */
   622→export function buildAirBSystemPrompt(
   623→  context: AirbnbPropertyContext,
   624→  intent: AirBIntent,
   625→  conversationHistory?: string,
   626→): string {
   627→  // ── Identidade do anfitrião ──
   628→  const identityBlock = `Você é o ANFITRIÃO do "${context.name}".
   629→Você NÃO é uma secretária, NÃO é um robô de atendimento, NÃO é um chatbot genérico.
   630→Você é como o DONO do imóvel que está recebendo um amigo em casa.
   631→Conhece CADA DETALHE do imóvel — onde fica a chave, como funciona o ar,
   632→qual a melhor padaria da rua, qual quarto tem o travesseiro mais macio.
   633→
   634→Tom: Pessoal, acolhedor, como um amigo que te empresta a casa.
   635→     NUNCA use linguagem corporativa ou de "atendimento ao cliente".
   636→     NUNCA diga "Como posso ajudar?" — soa robótico.
   637→     NUNCA tente vender nada — o hóspede JÁ reservou.
   638→
   639→Imóvel: ${context.name}
   640→Tipo: ${context.type}
   641→Localização: ${context.address}, ${context.neighborhood} — ${context.city}/${context.state}
   642→Capacidade: Até ${context.maxGuests} hóspedes`;
   643→
   644→  // ── Regras do anfitrião ──
   645→  const rulesBlock = `REGRAS DO ANFITRIÃO:
   646→1. Responda como se VOCÊ fosse o dono do imóvel — não como um atendente.
   647→2. Seja CONCISO — mensagens de WhatsApp longas cansam. Máximo 3 parágrafos curtos.
   648→3. Use emojis com moderação — como uma pessoa real, não um panfleto.
   649→4. Se não souber algo, diga "Deixa eu verificar isso pra você" — NÃO invente.
   650→5. NUNCA mencione preços, reservas, ou tente vender algo.
   651→6. NUNCA use "Estou aqui para ajudar" ou frases de call center.
   652→7. Emergências são PRIORIDADE — responda rápido e dê contatos claros.
   653→8. Sempre em português do Brasil, natural e coloquial.`;
   654→
   655→  // ── Conhecimento do anfitrião ──
   656→  const knowledgeBlock = buildHostKnowledgeBlock(context);
   657→
   658→  // ── Histórico da conversa ──
   659→  const historyBlock = conversationHistory
   660→    ? `\n\n=== HISTÓRICO DA CONVERSA ===\n${conversationHistory}`
   661→    : '';
   662→
   663→  // ── Diretrizes específicas por intenção ──
   664→  const intentBlock = buildIntentSpecificBlock(intent, context);
   665→
   666→  return `${identityBlock}\n\n${rulesBlock}\n\n${knowledgeBlock}${historyBlock}\n\n${intentBlock}`;
   667→}

   669→/**
   670→ * Constrói o bloco de conhecimento do anfitrião para o prompt.
   671→ * Inclui: peculiaridades do imóvel, equipamentos, regras, e dicas do bairro.
   672→ */
   673→function buildHostKnowledgeBlock(context: AirbnbPropertyContext): string {
   674→  const parts: string[] = ['=== O QUE VOCÊ SABE SOBRE O IMÓVEL ==='];
   675→
   676→  // Acesso
   677→  parts.push('\n🔑 Acesso:');
   678→  parts.push(`  Check-in: ${context.checkInInstructions}`);
   679→  if (context.lockCode) parts.push(`  Código/Chave: ${context.lockCode}`);
   680→  parts.push(`  WiFi: ${context.wifiNetwork} / Senha: ${context.wifiPassword}`);
   681→  if (context.parkingInstructions) parts.push(`  Estacionamento: ${context.parkingInstructions}`);
   682→
   683→  // Regras resumidas
   684→  parts.push('\n📜 Regras da Casa:');
   685→  for (const rule of context.houseRules.slice(0, 5)) {
   686→    parts.push(`  • ${rule}`);
   687→  }
   688→  if (context.quietHoursStart) {
   689→    parts.push(`  • Horário silencioso: ${context.quietHoursStart} às ${context.quietHoursEnd}`);
   690→  }
   691→
   692→  // Conhecimento do anfitrião (peculiaridades e avisos importantes)
   693→  const importantKnowledge = context.hostKnowledge.filter(k => k.isImportant || k.category === 'warning');
   694→  if (importantKnowledge.length > 0) {
   695→    parts.push('\n⚠️ Atenção (coisas que o hóspede precisa saber):');
   696→    for (const k of importantKnowledge) {
   697→      parts.push(`  • ${k.title}: ${k.description}`);
   698→    }
   699→  }
   700→
   701→  // Equipamentos resumidos
   702→  if (context.equipment.length > 0) {
   703→    parts.push('\n🔧 Equipamentos:');
   704→    for (const eq of context.equipment) {
   705→      parts.push(`  • ${eq.name} (${eq.location}): ${eq.instructions}`);
   706→    }
   707→  }
   708→
   709→  // Dicas do bairro (top 5)
   710→  if (context.neighborhoodTips.length > 0) {
   711→    parts.push('\n🗺️ Bairro:');
   712→    const topTips = context.neighborhoodTips.slice(0, 5);
   713→    for (const tip of topTips) {
   714→      parts.push(`  • ${tip.name}: ${tip.description}${tip.distance ? ` (${tip.distance})` : ''}`);
   715→    }
   716→  }
   717→
   718→  // Emergência
   719→  if (context.emergencyContacts.length > 0) {
   720→    parts.push('\n🚨 Emergência:');
   721→    for (const contact of context.emergencyContacts) {
   722→      parts.push(`  • ${contact.name} (${contact.role}): ${contact.phone}${contact.availableHours ? ` — ${contact.availableHours}` : ''}`);
   723→    }
   724→  }
   725→
   726→  return parts.join('\n');
   727→}

  1024→// ── 8. PIPELINE DE EXECUÇÃO (Prova de Conceito) ──────────────────────────────
  1025→
  1026→/**
  1027→ * Resultado do processamento do Zélla AirB.
  1028→ */
  1029→export interface AirBProcessResult {
  1030→  intent: AirBIntent;
  1031→  confidence: number;
  1032→  systemPrompt: string;
  1033→  userPrompt: string;
  1034→  toolResults: Array<{ tool: string; result: any }>;
  1035→  /** A resposta final seria gerada pelo LLM — aqui só montamos o prompt */
  1036→  readyForLLM: boolean;
  1037→}
  1038→
  1039→/**
  1040→ * Pipeline completo de processamento do Zélla AirB.
  1041→ *
  1042→ * Este é o equivalente ao chain de handlers do agent-orchestrator.ts:
  1043→ *   SecurityHandler → IntentClassifierHandler → TrialValidatorHandler →
  1044→ *   ReceiptHandler → PromptBuilderHandler → ToolCallingHandler →
  1045→ *   SemanticCacheHandler → LLMExecutionHandler → LoggingHandler → VoiceHandler
  1046→ *
  1047→ * Mas simplificado para a POC e com a lógica específica do AirB.
  1048→ *
  1049→ * No código real, este pipeline seria injetado no orquestrador
  1050→ * quando Property.type for um tipo AirB (apartamento|casa|studio|loft)
  1051→ * OU quando o tenant estiver no modo "airbnb".
  1052→ */
  1053→export async function processAirBMessage(
  1054→  message: string,
  1055→  context: AirbnbPropertyContext,
  1056→  conversationHistory?: string,
  1057→): Promise<AirBProcessResult> {
  1058→  // 1. Classificar intenção (equivalente ao IntentClassifierHandler)
  1059→  const { intent, confidence } = classifyAirBIntent(message);
  1060→
  1061→  // 2. Construir system prompt (equivalente ao PromptBuilderHandler)
  1062→  const systemPrompt = buildAirBSystemPrompt(context, intent, conversationHistory);
  1063→
  1064→  // 3. Construir user prompt
  1065→  const userPrompt = buildAirBUserPrompt(message, intent, confidence);
  1066→
  1067→  // 4. Executar tools relevantes (equivalente ao ToolCallingHandler)
  1068→  const toolResults = await executeAirBToolsForIntent(intent, context);
  1069→
  1070→  // 5. Montar resultado
  1071→  return {
  1072→    intent,
  1073→    confidence,
  1074→    systemPrompt,
  1075→    userPrompt,
  1076→    toolResults,
  1077→    readyForLLM: true,
  1078→  };
  1079→}
  1080→
  1081→/**
  1082→ * Executa as tools relevantes para a intenção detectada.
  1083→ *
  1084→ * No orquestrador atual (agent-orchestrator.ts:115-127),
  1085→ * o ToolCallingHandler sempre chama zehla_buscar_dados_property.
  1086→ * No AirB, chamamos as tools relevantes para a intenção.
  1087→ */
  1088→async function executeAirBToolsForIntent(
  1089→  intent: AirBIntent,
  1090→  context: AirbnbPropertyContext,
  1091→): Promise<Array<{ tool: string; result: any }>> {
  1092→  const tools = buildAirBTools(context);
  1093→  const results: Array<{ tool: string; result: any }> = [];
  1094→
  1095→  // Mapeamento intenção → tools
  1096→  const intentToolMap: Record<AirBIntent, string[]> = {
  1097→    CHECK_IN_GUIDE:          ['airb_get_checkin_guide'],
  1098→    SELF_CHECK_IN:           ['airb_get_checkin_guide', 'airb_get_wifi_info'],
  1099→    HOUSE_RULES:             ['airb_get_house_rules'],
  1100→    WIFI_INFO:               ['airb_get_wifi_info'],
  1101→    EQUIPMENT_HELP:          ['airb_get_equipment_help'],
  1102→    NEIGHBORHOOD_TIPS:       ['airb_get_neighborhood_tips'],
  1103→    PARKING_INFO:            ['airb_get_checkin_guide'],  // parking está no checkin guide
  1104→    EMERGENCY:               ['airb_get_emergency_info'],
  1105→    HOST_GREETING:           ['airb_get_checkin_guide'],
  1106→    HOST_FAREWELL:           [],
  1107→    EXTEND_STAY:             [],
  1108→    CLEANING_REQUEST:        [],
  1109→    MAINTENANCE_ISSUE:       ['airb_get_equipment_help', 'airb_get_host_knowledge'],
  1110→    LOCAL_RECOMMENDATION:    ['airb_get_neighborhood_tips', 'airb_get_host_knowledge'],
  1111→    HUMAN_HANDOVER:          [],
  1112→    UNKNOWN:                 ['airb_get_host_knowledge'],
  1113→  };
  1114→
  1115→  const toolNames = intentToolMap[intent] || [];
  1116→
  1117→  for (const toolName of toolNames) {
  1118→    const tool = tools.find(t => t.name === toolName);
  1119→    if (tool) {
  1120→      try {
  1121→        const result = await tool.execute({}, context);
  1122→        results.push({ tool: toolName, result: JSON.parse(result) });
  1123→      } catch (err) {
  1124→        results.push({ tool: toolName, result: { error: String(err) } });
  1125→      }
  1126→    }
  1127→  }
  1128→
  1129→  return results;
  1130→}

  1144→ *
  1145→ *   // No agent-orchestrator.ts

  1159→  buildSystemPrompt(
  1160→    property: AirbnbPropertyContext,

  1163→  ): string;
  1164→
  1165→  /** Constrói o user prompt */
  1166→  buildUserPrompt(
  1167→    message: string,
  1168→    intent: string,
  1169→    confidence: number,
  1170→    context?: Record<string, unknown>,
  1171→  ): string;
  1172→
  1173→  /** Retorna as tools disponíveis */
  1174→  getTools(property: AirbnbPropertyContext): AirBToolDefinition[];
  1175→
  1176→  /** Retorna o nome do agente para a intent */
  1177→  getAgentName(intent: string): string;
  1178→

  1180→  shouldIncludeSalesCTA(): boolean;
